cmake_minimum_required(VERSION 3.16)
project(circuits)

enable_testing()

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 17)

add_compile_options(-g -Wall -Wextra -Werror)

add_subdirectory(external/googletest)

add_test(complex_number_test complex_number_test)

add_test(graph_test graph_test)
add_test(input_parser_test input_parser_test)

add_test(big_circuit_test big_circuit_test)

add_executable(matrix_test t/matrix_test.cpp matrix.cpp matrix.h complex_number.cpp complex_number.h)

add_executable(big_circuit_test t/big_circuit_test.cpp circuit.cpp circuit.h input_parser.cpp input_parser.h graph.cpp graph.h matrix.cpp matrix.h complex_number.cpp complex_number.h)

target_link_libraries(complex_number_test gtest gmock gmock_main)
